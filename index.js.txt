const client = new Client({
  intents: [
    GatewayIntentBits.Guilds,
    GatewayIntentBits.GuildMessages
  ]

async function ensureHeart(thread) {
  try {
    const messages = await thread.messages.fetch({ limit: 1 });
    const firstMessage = messages.first();
    if (!firstMessage) return;

    const hasHeart = firstMessage.reactions.cache.some(
      r => r.emoji.name === HEART
    );

    if (!hasHeart) {
      await firstMessage.react(HEART);
      console.log(`â¤ï¸ Herz ergÃ¤nzt in Thread: ${thread.name}`);
    }
  } catch (err) {
    console.error(`âŒ Fehler in Thread ${thread.name}:`, err.message);
  }
}

  // ðŸ” Bestehende Threads prÃ¼fen

    for (const thread of activeThreads.threads.values()) {
      await ensureHeart(thread);
    }
  }
});
